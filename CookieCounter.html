  1 /**
  2  * CookieCounter class
  3  * This class keeps track of cookies sold by Girl Scout troops
  4  * each instance represents one troop
  5  * @author Kevin Liu
  6  * created on Dec 9 2013
  7  */
  8 public class CookieCounter {
  9     
 10     private int troop_number;               //troop number
 11     private static final int NUM_TYPES = 6; //number of types of cookies
 12     
 13     //Types of cookies and their corresponding numbers
 14     public static final int THIN_MINTS     = 1;
 15     public static final int SHORTBREAD     = 2;
 16     public static final int PEANUT_BUTTER  = 3;
 17     public static final int MACAROONS      = 4;
 18     public static final int LEMON_DROP     = 5;
 19     public static final int SNIKERDOODLES  = 6;
 20     
 21     //an array of strings that holds the name of cookies
 22     private static final String[] name = {"Thin Mints", "Shortbread", "Peanut butter", 
 23                                           "Macaroons", "Lemon drop", "Snikerdoodles"};
 24     
 25     //an array that holds the sales for each type of cookie
 26     private int[] sales = new int[NUM_TYPES];
 27     
 28     //an array that holds the sales for each type of cookies of all troops conbined
 29     private static int[] sales_all = new int[NUM_TYPES];
 30 
 31     /**
 32      * construct a troop instance and set troop number
 33      * @param troop_number troop number
 34      */
 35     public CookieCounter(int troop_number) {
 36         this.troop_number = troop_number;
 37     }
 38     
 39     /**
 40      * countSale is called when this troop sell out cookies
 41      * @param COOKIE_TYPE type of cookie
 42      * @param num number of boxes sold
 43      */
 44     public void countSale(int COOKIE_TYPE, int num) {
 45         if (validCookieNumber(COOKIE_TYPE)) {
 46             sales[COOKIE_TYPE-1] += num;   //because array index starts at 0, COOCIE_TYPE-1 is used
 47             sales_all[COOKIE_TYPE-1] += num;
 48         }else {
 49             System.out.print("\n");
 50             System.out.print(COOKIE_TYPE);
 51             System.out.println(" is not a valid cookie type!");
 52         }    
 53     }
 54 
 55     /**
 56      * displays sales of each type of cookie made by this troop
 57      */
 58     public void displayTroopSales() {
 59         //display troop number
 60         System.out.println("\n---Troop Sales---");
 61         System.out.print("Troop No.");
 62         System.out.println(troop_number);
 63         //display all cookie name and number of boxes sold
 64         for (int i=0; i<NUM_TYPES; i++) {
 65             System.out.print(name[i]+": \t");
 66             System.out.println(sales[i]);
 67         }
 68     }
 69     
 70     /**
 71      * displays sales of a certain type of cookie made by this troop
 72      * @param COOKIE_TYPE the type of cookie to be displayed
 73      */
 74     public void displayTroopSales(int COOKIE_TYPE) {
 75         if (validCookieNumber(COOKIE_TYPE)) {
 76         //display troop number
 77         System.out.println("\n---Troop Sales---");
 78         System.out.print("Troop No.");
 79         System.out.println(troop_number);
 80         
 81         //display cookie name and number of boxes sold
 82         System.out.print(name[COOKIE_TYPE-1]+": \t"); //because array index starts at 0, COOCIE_TYPE-1 is used
 83         System.out.println(sales[COOKIE_TYPE-1]);
 84         }else{
 85             System.out.print("\n");
 86             System.out.print(COOKIE_TYPE);
 87             System.out.println(" is not a valid cookie type!");
 88         }
 89     }
 90     
 91     /**
 92      * display sales for each type of cookie of all troops combined
 93      */
 94     public void displayAllSales() {
 95         System.out.println("\n---All Sales---");
 96         for (int i=0; i<NUM_TYPES; i++) {
 97             System.out.print(name[i]+": \t");
 98             System.out.println(sales_all[i]);
 99         }
100     }
101     
102     /**
103      * displays the bestselling type(s) of cookie
104      * if two or more types of cookie has the same
105      * sales, then all them are displayed
106      */
107     public void displayTroopBestseller() {
108         
109         int bestseller = 0;     //holds the biggest number in "sales" array
110         
111         //display troop number
112         System.out.println("\n---Troop Bestseller---");
113         System.out.print("Troop No.");
114         System.out.println(troop_number);
115         
116         //go through array "sales" and find the biggest number
117         for (int i=0; i<NUM_TYPES; i++) {
118             if (sales[i]>bestseller) {
119                 bestseller = sales[i];
120             }//end of if
121         }//end of for loop
122         
123         //go through array "sales" and "name" and print name and sales of bestseller(s)
124         for (int j=0; j<NUM_TYPES; j++) {
125             if (sales[j] == bestseller) {     
126                 System.out.print(name[j]+": \t");
127                 System.out.println(sales[j]);
128             }//end of if
129         }//end of for loop
130     }
131     
132     /**
133      * checks if a number is a valid cookie type number
134      * @param type_number the number to be checked
135      * @return true if number is a cookie type number, false if not
136      */
137     private boolean validCookieNumber(int type_number) {
138             return (type_number > 0) && (type_number <= NUM_TYPES);
139     }
140 
141 }//end of CookieCounter
